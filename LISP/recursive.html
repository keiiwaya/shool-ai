  1. フィボナッチ関数
    • f(n) = f(n - 1) + f(n - 2) for n > 1
    • f(0) = 1
    • f(1) = 1 

    >(defun fibonacci (n)
        (cond
            ((zerop n) 1)
            ((equal n 1) 1)
            (t (+ (fibonacci (1- n))
                  (fibonacci (- n 2))))))
FIBONACCI

>(fibonacci 4)

  3. memberp

    >(defun memberp (a lat)
        (cond
             ((null lat) nil)
             ((equal (car lat) a) t)
             (t (memberp a (cdr lat)))))

    >(defun count-item (lst item)
        (cond
             ((null lst) 0)
             ((equal (car lst) item) 
                (1+ (count-item (cdr lst) item)))
             (t (count-item (cdr lst) item))))

  5. remberp

    >(defun remberp (a lat)
        (cond
             ((null lat) '())
             ((equal (car lat) a) (remberp a (cdr lat)))
             (t (cons
                    (car lat)
                    (remberp a (cdr lat))))))
REMBERP

>(setq lst '(This is a list of atoms))

>(setq atm 'of)

  6. insertR

    >(defun  insertR (old new lat)
        (cond
             ((null lat) '())
             (t (cond
                    ((equal (car lat) old)
                         (cons old (cons new (cdr lat))))
                    (t (cons (car lat)
                             (insertR old new (cdr lat)))))))
INSERTR

>(setq lst '(This is a list of atoms))
(THIS IS A LIST OF ATOMS)

>(setq atmo 'of)
OF

>(setq atmn 'new)
NEW

>(insertR atmo atmn lst)
