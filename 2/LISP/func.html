関数定義

  1. defun 
    (defun 関数名　(パラメータ1, パラメータ2, ... パラメータn)
                プロセスの記述)

    >(defun  doll-to-yen (doll)
          (* doll 109))
DOLL-TO-YEN

>(setq book 15)
15

>(doll-to-yen book)
1635

  2. cond 
    (cond
     (テスト1　結果1)
     (テスト2　結果2)
              :
     (テストn　結果n))

    >(defun  reverse1 (lst switch)
       (cond
           ((equal 'yes switch) (reverse lst))
           (t lst)))
REVERSE1

>(setq lat '(I hate lisp))
(I HATE LISP)

>(reverse1 lat 'yes)
(LISP HATE I)

>(reverse1 lat 'no)
(I HATE LISP)
