  1. lambda

    局所的に(１回だけ)使う関数を、defunを使って名前を定めることなく、名前を使わないでlambadaを用いて関数を定義することができる。

    >(setq fruits
    '(orange apple pear apple grapefruit mango banana apple))
(ORANGE APPLE PEAR APPLE GRAPEFRUIT MANGO BANANA APPLE)

>(defun apple-1-0 (lst)
    
       (cond
           ((null lst) '())
           ((equal (car lst) 'apple)(cons 1 (apple-1-0 (cdr lst))))
           (t (cons 0(aplle-1-0 (cdr lst))))))
APPLE-1-0

>(apple-1-0 fruits)
(0 1 0 1 0 0 0 1)

>(apply #'+ (apple-1-0 fruits))
3

>(defun countapples (lst)
       (apply #'+ (apple-1-0 lst)))
COUNTAPPLES

>(countapples fruits)
3

>(defun countapples (lst)
       (apply #'+
               (mapcar '(lambda (x)
                          (cond
                              ((equal x 'apple) 1)
                              (t 0)))
                          lst)))
COUNTAPPLES

>(countapples fruits)
3
